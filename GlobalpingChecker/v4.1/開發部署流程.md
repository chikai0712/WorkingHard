#### 1.4 本機測試腳本

```bash
#!/bin/bash
# test-local.sh - 本機快速測試

echo "🧪 本機測試開始..."

# 1. 測試數據庫
echo "1️⃣ 測試數據庫..."
python test_db.py || exit 1

# 2. 測試 API（需要先啟動服務）
echo "2️⃣ 測試 API..."
curl -f http://127.0.0.1:8765/api/stats > /dev/null 2>&1 || {
    echo "❌ API 測試失敗"
    exit 1
}

# 3. 測試批次列表
echo "3️⃣ 測試批次列表..."
curl -f http://127.0.0.1:8765/api/batches > /dev/null 2>&1 || {
    echo "❌ 批次 API 測試失敗"
    exit 1
}

echo "✅ 本機測試通過！"
```

echo "2️⃣ 測試 Dashboard..."
curl -f http://$AWS_HOST:8000 > /dev/null 2>&1 || {
    echo "❌ Dashboard 無響應"

echo "3️⃣ 測試批次 API..."
curl -f http://$AWS_HOST:8000/api/batches > /dev/null 2>&1 || {
    echo "❌ 批次 API 無響應"
    exit 1

echo "✅ AWS 部署驗證通過！"
```

```

## 🔧 工具腳本

### 創建 Makefile

```makefile
# Makefile - 快速命令

.PHONY: test-local verify-aws deploy

test-local:
	@echo "🧪 本機測試..."
	@python test_db.py
	@curl -f http://127.0.0.1:8765/api/stats > /dev/null 2>&1 && echo "✅ API 正常" || echo "❌ API 失敗"

verify-aws:
	@echo "🔍 驗證 AWS..."
	@curl -f http://54.238.247.106:8000/api/stats > /dev/null 2>&1 && echo "✅ AWS 正常" || echo "❌ AWS 失敗"

deploy:
	@echo "🚀 部署到 AWS..."
	@./deploy-to-aws.sh

start-local:
	@echo "🏠 啟動本機服務..."
	@source venv/bin/activate && python -m uvicorn app.main:app --host 127.0.0.1 --port 8765 --reload
```

## 📊 測試數據管理

### 數據庫備份策略

```bash
# 本機數據庫（測試用）
data/globalping_results.db

# 備份
data/backups/
  ├── test_*.db          # 測試數據庫
  └── production_*.db    # 生產數據庫備份
```

### 切換測試/生產數據

```bash
# 使用測試數據
cp data/test_small.db data/globalping_results.db

# 使用生產數據
cp data/production_backup.db data/globalping_results.db
```
